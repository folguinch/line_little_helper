 22def _value_from_header(text: str):
 23    """Separate value and unit from spectrum header."""
 24    value = text.strip().split('=')[1].split()
 25    if value[0] == 'none':
 26        return None
 27    else:
 28        return float(value[0]) * u.Unit(value[1], format='cds')

 74    def __add__(self, spec):
 75        if self._frame != spec._frame:
 76            # This should be adapted in the future
 77            raise ValueError('Cannot append spectra in different frames')
 78        spectral_axis = np.append(self.spectral_axis, spec.spectral_axis)
 79        ind = np.argsort(spectral_axis)
 80        intensity = np.append(self.intensity, spec.intensity)
 81        if self.beam is not None:
 82            beam = np.append(self.beam, spec.beam)
 83            beam = beam.take(ind)
 84        else:
 85            beam = None
 86        return Spectrum(spectral_axis=spectral_axis[ind],
 87                        intensity=intensity[ind],
 88                        restfreq=self.restfreq,
 89                        restframe=self._frame,
 90                        vlsr=self.vlsr,
 91                        rms=self.rms,
 92                        beam=beam)

165    @classmethod
166    def from_file(cls, filename: Path,
167                  spectral_axis_unit: u.Unit = u.GHz) -> None:
168        """Load spectrum from `dat` file."""
169        # Read file
170        text = filename.read_text()
171        text = text.split('#')[1:]
172
173        # Get header values
174        restfreq = _value_from_header(text[0])
175        rms = _value_from_header(text[1])
176        vlsr = _value_from_header(text[2])
177        beam = text[3].split('=')[1]
178
179        # Frequency frame and units
180        restframe = 'rest' if 'Rest' in text[4].split()[0] else 'observed'
181        units = text[5].split()
182        units = tuple(map(lambda x: u.Unit(x, format='cds'), units))
183
184        # Get data
185        data = io.StringIO(text[6])
186        if beam == 'none':
187            spectral_axis, intensity = np.loadtxt(data, usecols=(0,2),
188                                                  unpack=True)
189            beam = None
190        else:
191            cols = (0, 2, 3, 4, 5)
192            spectral_axis, intensity, bmaj, bmin, pa = np.loadtxt(data,
193                                                                  usecols=cols,
194                                                                  unpack=True)
195
196            # Beam
197            beam = Beams(bmaj*units[3], bmin*units[4], pa*units[5])
198
199        return cls((spectral_axis * units[0]).to(spectral_axis_unit),
200                   intensity * units[2], restfreq = restfreq,
201                   restframe=restframe, vlsr=vlsr, rms=rms, beam=beam)

204                       bmaj: Optional[u.Quantity] = None,
205                       bmin: Optional[u.Quantity] = None) -> u.Quantity:
206        """Obtain the intensity axis in temperature units.

212        Returns:
213          The intensity axis in temperature units.

224        equiv = u.brightness_temperature(self.spectral_axis, beam_area=beam)
225        return self.intensity.to(u.K, equivalencies=equiv)

227    def _as_cassis(self) -> str:
228        """Generate an string with the spectrum in CASSIS format."""
229        fmt_str = ('%10s\t'*7).strip()
230        lines = [f'// number of line : {self.length - 1}']
231        if self.vlsr is not None and self._frame == 'observed':
232            lines.append(f'// vlsr : {-self.vlsr.value:.1f}')
233        else:
234            lines.append('// vlsr : 0.0')
235        lines.append(fmt_str % ('FreqLsb', 'VeloLsb', 'FreqUsb', 'VeloUsb',
236                                'Intensity', 'DeltaF', 'DeltaV'))
237
238        fmt_tab = ('%10.3f\t' + '%10.4f\t'*6).strip()
239        delta_nu = np.abs(np.diff(self.spectral_axis.to(u.MHz).value))
240        delta_vel = np.abs(np.diff(self.velocity_axis.to(u.m/u.s).value))
241        zeros = np.zeros(delta_nu.shape)
242        data = (self.spectral_axis.to(u.MHz).value,
243                self.velocity_axis.to(u.m/u.s).value,
244                zeros,
245                zeros,
246                self.to_temperature().value,
247                delta_nu,
248                delta_vel)
249        lines += [fmt_tab % d for d in zip(*data)]
250
251        return '\n'.join(lines)

253    def _as_dat(self) -> str:
254        """Generate an string with the spectrum in tab-separated format."""
255        # Header
256        if self.restfreq is None:
257            header = '#restfreq=none\n'
258        else:
259            header = f'#restfreq={self.restfreq.value} {self.restfreq.unit}\n'
260        if self.rms is None:
261            header += '#rms=none\n'
262        else:
263            header += f'#rms={self.rms.value} {self.rms.unit}\n'
264        if self.vlsr is None:
265            header += '#vlsr=none\n'
266        else:
267            header += f'#vlsr={self.vlsr.value} {self.vlsr.unit}\n'
268        freq = 'RestFreq' if self._frame == 'rest' else 'ObsFreq'
269
270        # Data and units
271        data = (self.frequency_axis.value,
272                self.velocity_axis.value,
273                self.intensity.value)
274        units = (f'#{self.frequency_axis.unit:cds}'
275                 f' {self.velocity_axis.unit:cds}'
276                 f' {self.intensity.unit:cds}')
277        if self.beam is None:
278            header += '#beam=none\n'
279            header += f'#{freq} Vel Flux'
280        else:
281            header += '#beam=true\n'
282            header += f'#{freq} Vel Flux bmaj bmin pa'
283            try:
284                if len(self.beam) == self.length:
285                    data += (self.beam.major.value,
286                             self.beam.minor.value,
287                             self.beam.pa.value)
288                else:
289                    raise ValueError('Beam array with wrong size')
290            except TypeError:
291                data += (np.repeat(self.beam.major.value, self.length),
292                         np.repeat(self.beam.minor.value, self.length),
293                         np.repeat(self.beam.pa.value, self.length))
294            units += (f' {self.beam.major.unit:cds}'
295                      f' {self.beam.minor.unit:cds}'
296                      f' {self.beam.pa.unit:cds}')
297        fmt_tab = '%f\t' * len(data)
298        fmt_tab = fmt_tab.strip()
299        units += '#'
300        lines = [fmt_tab % d for d in zip(*data)]
301
302        return '\n'.join([header, units] + lines)

320    def range_mask(self,
321                   low: Optional[u.Quantity] = None,
322                   up: Optional[u.Quantity] = None) -> np.array:
323        """Creates a mask from spectral axis limits."""
324        mask = np.ones(self.spectral_axis.shape, dtype=bool)
325        if low is not None:
326            mask = self.spectral_axis >= low
327        if up is not None:
328            mask = mask & (self.spectral_axis <= up)
329
330        return mask

334        if self.rms is not None:

351    def has_lines(self, min_width: int = 5,
352                  dilate: Optional[int] = None) -> list:
353        """Find ranges with likely line emission.
354
355        Args:
356          min_width: optional; minimum number of channels to be considered a
357            line.
358          dilate: optional; number of channels to add around the lines.
359
360        Returns:
361          A list of slices.
362        """
363        # Generate mask
364        mask = self.intensity_mask()
365
366        # No lines
367        if np.sum(mask) == 0:
368            return []
369
370        # First pass
371        # Labels and objects
372        labs, nlabs = ndimage.label(mask)
373        objs = ndimage.find_objects(labs)
374        # Filter slices
375        for obj in objs:
376            slc = obj[0]
377            delta = slc.stop - slc.start
378            if delta > min_width:
379                continue
380            else:
381                mask[slc] = False
382
383        # Erode mask
384        if dilate is not None:
385            mask = ndimage.binary_dilation(mask, iterations=dilate)
386
387        # Final objects
388        labs, nlabs = ndimage.label(mask)
389        objs = ndimage.find_objects(labs)
390
391        return objs

393    def fit_line(self, spec_range: Optional[Sequence[u.Quantity]] = None,
394                 slice_range: Optional[slice] = None) -> models.Gaussian1D:
395        """Fit the spectrum with a Gaussian funtion.
396
397        Args:
398          spec_range: optional; upper and lower limits of the spectral axis.
399          slice_range: optional; slice object of the data to fit.
400
401        Returns:
402          A Gaussian model with the fitted parameters.
403        """
404        # Select data
405        if spec_range is not None:
406            mask = self.range_mask(*spec_range)
407            x = self.spectral_axis[mask]
408            y = self.intensity[mask]
409        elif slice_range is not None:
410            x = self.spectral_axis[slice_range]
411            y = self.intensity[slice_range]
412        else:
413            x = self.spectral_axis
414            y = self.intensity
415
416        # Initial conditions
417        aux = x[y > np.max(y)/2]
418        stddev = np.abs(aux[-1] - aux[0]) / 2
419        model = models.Gaussian1D(amplitude=np.max(y), mean=x[len(x)//2],
420                                  stddev=stddev)
421
422        # Fitter
423        fitter = fitting.LevMarLSQFitter()
424        model_fit = fitter(model, x, y)
425
426        return model_fit

428    def fit_lines(self, min_width: int = 5,
429                  dilate: Optional[int] = None,
430                  slice_as_freq: bool = False,
431                  ) -> Tuple[List[slice], List[models.Gaussian1D]]:
432        """Fit all the potential lines.
433
434        Args:
435          min_width: optional; minimum number of channels to be considered a
436            line.
437          dilate: optional; number of channels to add around the lines.
438          slice_as_frequency: optional; convert the index slices to frequency
439            range?
440
441        Returns:
442          The slices where the lines are fit.
443          The model results.
444        """
445        # Find slices with lines
446        slices = self.has_lines(min_width=min_width, dilate=dilate)
447
448        # Fit each slice
449        results = []
450        final_slices = []
451        for slc in slices:
452            results.append(self.fit_line(slice_range=slc))
453            if slice_as_freq:
454                final_slices = [(self.frequency_axis[slc.start],
455                                 self.frequency_axis[slc.stop-1])]
456
457        # Check slices
458        if len(final_slices) == 0:
459            final_slices = slices
460
461        return final_slices, results

463    def peak_frequency(self,
464                       low: Optional[u.Quantity] = None,
465                       up: Optional[u.Quantity] = None) -> u.Quantity:
466        """Return the spectral axis value at the intensity peak.
467
468        Regions of the spectral axis where to find the peak can be limited with
469        the low and up parameters.
470
471        Args:
472          low: optional; lower spectral axis limit.
473          upper: optional; upper spectral axis limit.
474        Returns:
475          The spectral axis value where intensity is max.
476        """
477        mask = self.range_mask(low=low, up=up)
478        ind = np.nanargmax(self.intensity[mask])
479
480        return self.spectral_axis[mask][ind]

482    def centroid(self,
483                 low: Optional[u.Quantity] = None,
484                 up: Optional[u.Quantity] = None,
485                 nsigma: float = 5) -> u.Quantity:
486        """Determine the spectral axis value center of mass.
487
488        Args:
489          low: optional; lower spectral axis limit.
490          upper: optional; upper spectral axis limit.
491          nsigma: optional; lower limit for the emission.
492        Returns:
493          The spectral axis value weighted by intensity.
494        """
495        mask = self.combined_mask(low=low, up=up, nsigma=nsigma)
496        cm = np.sum(self.spectral_axis[mask] * self.intensity[mask])
497        cm = cm / np.sum(self.intensity[mask])
498
499        return cm.to(self.spectral_axis.unit)

623    def from_arrays(
624        cls,
625        arrays: List,
626        restfreq: u.Quantity,
627        equivalencies: dict,
628        vlsr: Optional[u.Quantity] = None,
629        rms: Optional[List[u.Quantity]] = None,
630        freq_names: Sequence[str] = ('nu', 'freq', 'frequency',
631                                     'v', 'vel', 'velocity'),
632        flux_names: Sequence[str] = ('F', 'f', 'Fnu', 'fnu',
633                                     'intensity', 'T', 'Tb'),
634    ):
635        """Creates spectra object from a list structured array.
636
637        The values in `arrays` are pairs consisting of an structured `np.array`
638        and a dictionary with the units for each column.
639        """
640        specs = []
641        for spw, (data, units) in enumerate(arrays):
642            # Spectral axis
643            freq_name = list(filter(lambda x, un=units: x in un, freq_names))[0]
644            xaxis = data[freq_name] * units[freq_name]
