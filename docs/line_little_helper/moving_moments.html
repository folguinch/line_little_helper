#!/bin/bash
"""Find line velocity at peak and calculate 1st moment maps incresing the
number of channels.
"""
import argparse
import os
import sys

from astropy.io import fits
from astropy.stats import gaussian_fwhm_to_sigma
from toolkit.math import quick_rms
import astropy.units as u
import toolkit.argparse_tools.actions as actions
import toolkit.argparse_tools.parents as parents
import toolkit.argparse_tools.loaders as aploaders
import numpy as np
import scipy.ndimage as ndimg

def get_header(cube, bunit=None):
    # Header
    header = cube.wcs.sub(['longitude', 'latitude']).to_header()

    # Unit
    if bunit is not None:
        header['BUNIT'] = '{0:FITS}'.format(bunit)

    # Copy other cards
    cards = ['BMIN', 'BMAJ', 'BPA']
    for card in cards:
        if card in cube.header:
            header[card] = cube.header[card]

    return header

def save_mask(mask, header, filename, dtype=int):
    hdu = fits.PrimaryHDU(mask.astype(dtype))
    hdu.header = header
    hdu.writeto(filename, overwrite=True)

def get_results(vel, data, cube, outdir, logger, savemasks=False):
    # Max index along spectral axis
    mask = np.all(data.mask, axis=0)
    maxind = np.nanargmax(data, axis=0)
    maxvel = vel[maxind]
    maxvel[mask] = np.nan

    # Save
    output = os.path.join(outdir[0], 'line_%s.fits')
    hdu = fits.PrimaryHDU(maxind)
    hdu.header = get_header(cube)
    hdu.writeto(output % 'spec_peak_index', overwrite=True, output_verify='fix')
    hdu = fits.PrimaryHDU(maxvel.value)
    hdu.header = get_header(cube, bunit=vel.unit)
    hdu.writeto(output % 'spec_peak_vel', overwrite=True, output_verify='fix')

    # FHWM and stddev in channels
    data_filled = np.ma.filled(data, fill_value=np.nan)
    mask = data_filled > 0.5*np.nanmax(data_filled, axis=0)
    mask[data.mask] = False
    if savemasks:
        filename = output % ('fwhm_mask')
        save_mask(mask, cube.header, filename, dtype=float)
    fwhm_chan = np.sum(mask, axis=0)
    fwhm_chan = np.ma.array(fwhm_chan, mask=np.all(data.mask, axis=0))
    sigma = gaussian_fwhm_to_sigma * fwhm_chan
    medchan = int(np.ma.median(sigma))
    meanchan = int(np.ma.mean(sigma))
    logger.info('Median line stddev: %i channels', medchan)
    logger.info('Mean line stddev: %i channels', meanchan)
    if medchan==0:
        logger.warn('Median linewidth=0, using 1 instead')
        medchan = 1

    # Iterate over sigma values
    binstruc = np.zeros((3,)*3, dtype=bool)
    binstruc[:, 1, 1] = True
    m, n = maxind.shape
    I, J = np.ogrid[:m,:n]
    imgs = []
    for i in [1, 2, 3, 5, 8, 10]:
        # Auxiliary mask
        aux = np.zeros(data.shape, dtype=bool)

        # Set max values to True
        aux[maxind, I, J] = True
        aux[0][np.all(data.mask, axis=0)] = False
        if i==1 and savemasks:
            filename = output % ('moment1_0sigma_mask')
            save_mask(aux, cube.header, filename)

        # Dilate mask
        logger.info('Dilating %i times', i*medchan)
        aux = ndimg.binary_dilation(aux, structure=binstruc, 
                iterations=i*medchan)
        aux = aux & ~data.mask
        if savemasks:
            filename = output % ('moment1_%isigma_mask' % i)
            save_mask(aux, cube.header, filename)

        # Subcube
        aux = cube.with_mask(aux)
        aux = aux.moment(order=1)
        aux.write(output % ('moment1_%isigma' % i), overwrite=True)

def loader(args):
    # Load cube
    args.log.info('Loading cube')
    args.cube(args, args.cubename)

    # Load config
    args.log.info('Loading configuration')
    args.config(args, args.lineconfig)

def proc(args):
    # Create subcube
    chan1, chan2 = args.config.getintlist(args.section[0],'chanran')
    args.log.info('Channel range: %i, %i', chan1, chan2)
    restfreq = args.config.getquantity(args.section[0],'freq')
    subcube = args.cube[chan1:chan2+1,:,:]
    subcube = subcube.with_spectral_unit(u.km/u.s, 
            velocity_convention='radio',
            rest_value=restfreq)

    # Create cube mask
    if 'RMS' in subcube.header:
        args.log.info('RMS key in cube header')
        rms = float(subcube.header['RMS'])*subcube.unit
    else:
        args.log.info('Calculating rms value')
        rms = quick_rms(subcube)*subcube.unit
    args.log.info('rms value: {0.value} {0.unit}'.format(rms))
    args.log.info('Filtering out data < %i rms', args.nsigma)
    mask = subcube > rms*args.nsigma
    subcube = subcube.with_mask(mask)
    if args.shrink:
        subcube = subcube.minimal_subcube()
    data = subcube.filled_data[:].to(rms.unit).value
    data = np.ma.array(data, mask=np.isnan(data))

    # Velocity axis
    vel = subcube.spectral_axis

    # Calculate results
    get_results(vel, data, subcube, args.outdir, args.log,
            savemasks=args.savemasks)

def main(args):
    """Moment 1 maps from different windows.
    
    Args:
      args: arguments for argparse.
    """
    # Parser
    pipe = [loader, proc]
    args_parents = [parents.logger('debug_moving_moments.log')]
    parser = argparse.ArgumentParser(
        add_help=True,
        description='Moment 1 maps from incresing local window size',
        formatter_class=argparse.ArgumentDefaultsHelpFormatter,
        parents=args_parents)
    parser.add_argument('--shrink', action='store_true',
                        help='Shrink to minimal cube')
    parser.add_argument('--savemasks', action='store_true',
                        help='Save masks at each step')
    group1 = parser.add_mutually_exclusive_group(required=True)
    #parser.add_argument('lineconfig', nargs=1, action=actions.CheckFile,
    #                    help='Line configuration file')
    #parser.add_argument('section', nargs=1,
    #                    help='Configuration section')

    parser.add_argument('cubename', nargs=1, action=actions.CheckFile,
                        help='Cube file name')
    parser.add_argument('outdir', nargs=1, action=actions.MakePath,
                        help='Output directory')
    parser.set_defaults(pipe=pipe, results=None,
            cube=aploaders.load_spectral_cube,
            config=aploaders.load_config, nsigma=5.)
    args = parser.parse_args(args)
    for fn in args.pipe:
        fn(args)

if __name__=='__main__':
    main(sys.argv[1:])
