 14    """Class to store line information.
 15
 16    Attributes:
 17      species: chemical formula.
 18      qns: quantum numbers.
 19      restfreq: rest frequency.
 20      obsfreq: observed frequency.
 21      eup: upper level energy.
 22      logaij: log of the Aij coeficient.
 23    """
 24
 25    def __init__(self, species: str, qns: str, restfreq: u.Quantity,
 26                 obsfreq: Optional[u.Quantity] = None,
 27                 vlsr: Optional[u.Quantity] = None,
 28                 eup: Optional[u.Quantity] = None,
 29                 logaij: Optional[u.Quantity] = None) -> None:
 30        """Initiate a transition object.
 31
 32        Args:
 33          species: chemical formula.
 34          qns: quantum numbers.
 35          restfreq: transition rest frequency.
 36          obsfreq: optional; observed frequency.
 37          vlsr: optional; used to determine `obsfreq` if not given.
 38          eup: optional; upper level energy.
 39          logaij: optional; log of the Aij coeficient.
 40        """
 41        self.species = species
 42        self.qns = qns
 43        self.restfreq = restfreq.to(u.GHz)
 44        self.obsfreq = obsfreq
 45        if self.obsfreq is None and vlsr is not None:
 46            equiv = u.doppler_radio(self.restfreq)
 47            self.obsfreq = to_rest_freq(self.restfreq, -vlsr, equiv)
 48        try:
 49            self.obsfreq = self.obsfreq.to(u.GHz)
 50        except AttributeError:
 51            pass
 52        self.eup = eup
 53        self.logaij = logaij
 54
 55    def __str__(self):
 56        lines = [f'Species: {self.species}', f'QNs: {self.qns}']
 57        lines.append(f'Rest Freq = {self.restfreq.value} {self.restfreq.unit}')
 58        if self.obsfreq is not None:
 59            lines.append(f'Obs Freq = {self.obsfreq.value} {self.obsfreq.unit}')
 60        if self.eup is not None:
 61            lines.append(f'Eup = {self.eup.value} {self.eup.unit}')
 62        if self.logaij:
 63            lines.append(f'log(Aij) = {self.logaij}')
 64        return '\n'.join(lines)

 80            vlsr = config.getquantity('vlsr', fallback=None)
 81        eup = config.getquantity('eup', fallback=None)
 82        logaij = config.getfloat('logaij', fallback=None)

 84        return cls(species, qns, restfreq, obsfreq=obsfreq, eup=eup, vlsr=vlsr,
 85                   logaij=logaij)

 87    def generate_name(
 88        self,
 89        include: Optional[Sequence[str]] = ('species', 'qns'),
 90    ) -> str:
 91        """Generate a name string from requested keys.
 92
 93        The default is to use the species and QNs for the name. It replaces any
 94        brackets and comas with underscores.
 95
 96        Args:
 97          include: optional; which values include in the name.
 98        """
 99        name = []
100        if 'species' in include:
101            name.append(f'{self.species}'.replace('=', ''))
102        if 'qns' in include:
103            qns = f'{self.qns}'
104            qns = qns.replace('-', '_').replace('(', '').replace(')', '')
105            qns = qns.replace(',', '_').replace('=','_').replace('/', '_')
106            name.append(qns)
107        return '_'.join(name)

155    def from_query(cls, name: str, freq_range: QPair,
156                   vlsr: Optional[u.Quantity] = None,
157                   filter_out: Optional[list] = None,
158                   qns: Optional[str] = None,
159                   **kwargs):
160        """Obtain information about molecule from splat.
161
162        Args:
163          name: species name following
164            [astroquery](https://astroquery.readthedocs.io/en/v0.1-0/splatalogue.html).
165          freq_range: frequency range to look for transitions.
166          vlsr: optional; LSR velocity to obtain observed frequency.
167          filter_out: optional; filter out transitions with given QNs.
168          kwargs: optional additional query constraints.
169        """
170        transitions = _query_to_transition(name, freq_range, vlsr=vlsr,
171                                           filter_out=filter_out, qns=qns,
172                                           **kwargs)
173        return cls(name.strip(), transitions)

175    def reduce_qns(self):
176        """Delete repeated QNs."""
177        qns = self.qns_list
178        qns_set = set(qns)
179        if len(qns) != len(qns_set):
180            new_tra = []
181            for qn, tra in zip(qns, self.transitions):
182                if qn in qns_set:
183                    new_tra.append(tra)
184                    qns_set.remove(qn)
185            self.transitions = new_tra

197def _query_to_transition(name: str,
198                         freq_range: QPair,
199                         vlsr: Optional[u.Quantity] = None,
200                         filter_out: Optional[list] = None,
201                         qns: Optional[str] = None,
202                         #interactive: bool = False,
203                         #ntransitions: Optional[int] = None,
204                         **kwargs) -> List[Transition]:
205    """Perform a query, filter results and create transitions."""
206    # Query
207    table = query_lines(freq_range, chemical_name=name, **kwargs)
208    table['Freq'] = combine_columns(table, ['Freq', 'MeasFreq'])
209
210    # Filter QNs
211    if qns is not None:
212        ind = table['QNs'] == qns
213        try:
214            table = table[ind]
215        except IndexError as exc:
216            print('Could not find QNs')
217            print('Available QNs:')
218            print(table['QNs'])
219            raise ValueError('No QNs detected') from exc
220
221    # Create transitions
222    transitions = []
223    cols = ['Species', 'QNs', 'Freq', 'Eu', 'log10Aij']
224    for data in zip_columns(table, cols):
225        # Filter transitions
226        if filter_out is not None:
227            aux = False
228            for fil in filter_out:
229                if fil in data[1]:
230                    aux = True
231                    break
232            if aux:
233                continue
234
235        # Do not use unresolved QNs
236        if str(data[1]) == '--':
237            continue
238        transitions.append(
239            Transition(data[0], data[1], data[2],
240                       vlsr=vlsr, eup=data[3], logaij=data[4]))
241
242    return transitions
